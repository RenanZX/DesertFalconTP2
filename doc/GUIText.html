def initialize(texto, posX, posY, color = Gosu::Color::WHITE, tamfont = 20, espec = 1)
  tamfont = 20 if tamfont > 40 || tamfont < 10

  @font = Gosu::Font.new(tamfont)
  @valor = texto
  @color = color
  @posx = posX
  @posy = posY
  @posz = 0
  @especura = espec
end
