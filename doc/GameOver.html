def update
  if @continue.color == Gosu::Color::WHITE
    @continue.color = Gosu::Color::BLACK
  elsif @continue.color == Gosu::Color::BLACK
    @continue.color = Gosu::Color::WHITE
  end
  if Gosu.button_down?(Gosu::KB_ENTER) || Gosu.button_down?(Gosu::KB_RETURN)
    sleep 0.15
    return true
  end
end
