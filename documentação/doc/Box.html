def overlaps_with(gameobj)
  if !gameobj.nil? && !gameobj.box.nil?
    other_box = gameobj.box
    if @x < other_box.x + other_box.w &&
       @x + @w > other_box.x &&
       @y < other_box.y + other_box.h &&
       @h + @y > other_box.y
      return true
    end
  end
  false
end
